## Primary font family

The primary font family is used for most text on the page. It is set to Helvetica, "Helvetica Neue", Arial, sans-serif.

{
  Object.entries(theme.font.weight).map(([key, value]) => {
    return (
      <div key={key}>
        <h3>{key.charAt(0).toUpperCase() + key.slice(1)}</h3>
        <Typeset
          fontSizes={Object.values(theme.font.size).map((size) => size )}
          fontWeight={value} 
          sampleText="At University College London, the quick brown fox jumps over the lazy dog."
          fontFamily={theme.font.family.primary}
        />
      </div>
    );
  })
}
